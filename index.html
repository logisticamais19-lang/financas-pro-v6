async function sincronizarMesDoFirebase(){ if(!usuarioAtual) return; const d = getMes(mesAtual); // Limpa dados locais antes de sincronizar d.contas = []; d.diarios = []; d.cartoes.forEach(c => c.gastos = []); try { // 🔹 CONFIG MENSAL (salário + meta) const configDocId = `${usuarioAtual.username}_${mesAtual}`; const configSnap = await window.fbGetDoc( window.fbDoc(window.dbFirebase, "config_mensal", configDocId) ); if(configSnap.exists()){ const cfg = configSnap.data(); if(cfg.salario !== undefined) d.salario = cfg.salario; if(cfg.meta !== undefined) d.meta = cfg.meta; console.log(`✅ Config ${mesAtual} carregada`); } // 🔹 CONTAS FIXAS const contasSnap = await window.fbGetDocs( window.fbQuery( window.fbCollection(window.dbFirebase, "contas_fixas"), window.fbWhere("usuario", "==", usuarioAtual.username), window.fbWhere("mes", "==", mesAtual) ) ); contasSnap.forEach(docSnap => { const data = docSnap.data(); d.contas.push({ id: docSnap.id, descricao: data.descricao, emoji: data.emoji, valor: data.valor, dia: data.dia, pago: data.pago }); }); // 🔹 GASTOS DIÁRIOS const diariosSnap = await window.fbGetDocs( window.fbQuery( window.fbCollection(window.dbFirebase, "gastos_diarios"), window.fbWhere("usuario", "==", usuarioAtual.username), window.fbWhere("mes", "==", mesAtual) ) ); diariosSnap.forEach(docSnap => { d.diarios.push({ id: docSnap.id, ...docSnap.data() }); }); // 🔹 GASTOS CARTÃO const cartaoSnap = await window.fbGetDocs( window.fbQuery( window.fbCollection(window.dbFirebase, "gastos_cartao"), window.fbWhere("usuario", "==", usuarioAtual.username), window.fbWhere("mes", "==", mesAtual) ) ); cartaoSnap.forEach(docSnap => { const data = docSnap.data(); const cartao = d.cartoes.find(c => c.id == data.cId); if(cartao){ cartao.gastos.push({ id: docSnap.id, data: data.data, forma: data.forma, desc: data.desc, valor: data.valor, rotulo: data.rotulo, nota: data.nota }); } }); saveDB(); console.log("✅ Sincronização mensal concluída"); } catch(e){ console.error("Erro ao sincronizar:", e); } }
